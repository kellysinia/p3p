namespace DanielCrawl { class Program { static void Main(string[] args) { var result = DanielCrawl.GetJSDataStructures("http://webserv.nhl.nl/~oosterha/eng2014/index.php?sfpg=SW5mb3JtYXRpY2EvaTFhLyoqMjY2MmQ2NzNiOTJjOTYwMTAzMTUyNzBlM2Q1NzU0OWQ"); foreach (KeyValuePair> array in result) { Console.WriteLine("===== {0} =====", array.Key); foreach (KeyValuePair entry in array.Value) Console.WriteLine("{0}: {1}", entry.Key, entry.Value); Console.WriteLine("\n"); } Console.ReadKey(); } } class DanielCrawl { public static Dictionary> GetJSDataStructures(string url) { Dictionary> output = new Dictionary>(); string content; using (WebClient client = new WebClient()) content = client.DownloadString(url); foreach(Match m in Regex.Matches(content, @"(\w+)\[(\d+)\]\s*=\s*[\'](.*)[\'];")) { if (!output.ContainsKey(m.Groups[1].Value)) output.Add(m.Groups[1].Value, new Dictionary()); int index = Convert.ToInt32(m.Groups[2].Value); output[m.Groups[1].Value].Add(index, m.Groups[3].Value); } return output; } } }
